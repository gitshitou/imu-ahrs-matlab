1function y = ekf_6dof(gyroX,gyroY,gyroZ,accX,accY,accZ,dt ,TX)
 2%function y = test_ekf(gyroX,gyroY,gyroZ,accX,accY,accZ,dt )
 3%     gyroX,gyroY,gyroZ,...        %陀螺仪原始值
 4%     accX,accY,accZ,...         %加速度原始值 m/s^2
 5%     %magX,magY,magZ,...         %地磁原始值
 6%     dt)  %积分时间
 7 
 8%状态转移矩阵..。
 9A= [1  , -0.5*dt*gyroX   ,-0.5*dt*gyroY  , -0.5*dt*gyroZ;...
10    0.5*dt*gyroX , 1  ,  0.5*dt*gyroZ  ,  -0.5*dt*gyroY;...
11    0.5*dt*gyroY , -0.5*dt*gyroZ , 1  ,   0.5*dt*gyroX;...
12    0.5*dt*gyroZ ,   0.5*dt*gyroY ,   -0.5*dt*gyroX,  1];
13%Q
14%R
15%--- Dataset parameters
16%deltat = 1/200;             % Sampling period
17%noise_gyro = 2.4e-2;        % Gyroscope noise(discrete), rad/s
18%noise_accel = 2.83e-2;      % Accelerometer noise, m/s^2
19
20noise_gyro = 2.4e-3;        % Gyroscope noise(discrete), rad/s
21noise_accel = 2.83e-2; 
22gravity = 9.81007;          % Gravity magnitude, m/s^2
23
24persistent states_est P_P_est                % Initial state conditions
25if isempty(states_est)
26    %states_est = [1;0;0;0];             % x_est=[x,y,Vx,Vy,Ax,Ay]'
27    states_est = TX; 
28    P_P_est = 1e-10 * eye(4);
29end
30
31% 状态预测，协方差预测
32states_prd = A * states_est;
33states_prd = states_prd / norm(states_prd);    %四元数归一化
34G = [-states_est(2)  -states_est(3)  -states_est(4); ...
35          states_est(1)  -states_est(4)   states_est(3); ...
36          states_est(4)   states_est(1)  -states_est(2); ...
37         -states_est(3)   states_est(2)   states_est(1)] / 2;
38Q = (noise_gyro * dt)^2 * (G * G');
39P_P_prd = A * P_P_est * A' + Q;
40
41%估计量
42a =[accX; accY; accZ];
43% [ -2*q2,  2*q3, -2*q0, 2*q1]
44% [  2*q1,  2*q0,  2*q3, 2*q2]
45% [  2*q0, -2*q1, -2*q2, 2*q3]
46H =[ -2*states_prd(3), 2*states_prd(4), -2*states_prd(1), 2*states_prd(2);...
47      2*states_prd(2), 2*states_prd(1),  2*states_prd(4), 2*states_prd(3);...
48      2*states_prd(1), -2*states_prd(2), -2*states_prd(3),2*states_prd(4)];
49% H =[ (98*states(3))/5, -(98*states(4))/5,  (98*states(1))/5, -(98*states(2))/5;...
50%     -(98*states(2))/5, -(98*states(1))/5, -(98*states(4))/5, -(98*states(3))/5;...
51%     -(98*states(1))/5,  (98*states(2))/5,  (98*states(3))/5, -(98*states(4))/5];
52innvo = a/norm(a) - H*states_prd;
53% Measurement noise R
54R_internal = (noise_accel / norm(a))^2 * eye(3);
55R_external = (1-gravity/norm(a))^2 * eye(3);
56R = R_internal + R_external;
57
58Kg = P_P_prd * H'/(H*P_P_prd*H'+R);                   %Kg= P(k|k-1)HT/(HP(k|k-1)HT+R)
59states_est = states_prd + Kg* innvo;
60%P_P_est = P_P_prd -P_P_prd * H;    eye(4)
61P_P_est = (eye(4) - Kg * H) * P_P_prd;
62y = states_est;
63end
